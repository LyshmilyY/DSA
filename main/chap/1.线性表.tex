\chapter{线性表}

\begin{definition}[线性表]
    $$L=(a_{1},a_{2},a_{3},\dots,a_{i},a_{i+1},\dots,a_{n})$$
    \begin{enumerate}
        \item 有相同数据类型的 $n$ 个数据元素的有限序列
        \item 需要在原数据上进行修改
        \item 初始化、插入、删除(传地址); 长度、判空、打印(传值)
    \end{enumerate}
\end{definition}
